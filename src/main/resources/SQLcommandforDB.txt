CREATE DATABASE telephonica
    WITH
    OWNER = root
    ENCODING = 'UTF8'
    CONNECTION LIMIT = -1;

  drop table if exists t_services cascade
  drop sequence if exists hibernate_sequence

    create table t_services (
       id int8 not null,
        cost float8 check (cost>=0),
        description varchar(255),
        name varchar(255),
        primary key (id)
    )
